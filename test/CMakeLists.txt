## MD5

ADD_TEST(md5_version_short ${PROJECT_BINARY_DIR}/md5sum -v)
SET_TESTS_PROPERTIES(md5_version_short PROPERTIES PASS_REGULAR_EXPRESSION "md5sum for OS X [0-9].[0-9]")

ADD_TEST(md5_version_long ${PROJECT_BINARY_DIR}/md5sum --version)
SET_TESTS_PROPERTIES(md5_version_long PROPERTIES PASS_REGULAR_EXPRESSION "md5sum for OS X [0-9].[0-9]")

ADD_TEST(md5_help ${PROJECT_BINARY_DIR}/md5sum --help)
SET_TESTS_PROPERTIES(md5_help PROPERTIES PASS_REGULAR_EXPRESSION "Usage: md5sum file1 file2")

ADD_TEST(md5_will_sum_mismatch ${PROJECT_BINARY_DIR}/md5sum -c test_sum_mismatch.md5)
SET_TESTS_PROPERTIES(md5_will_sum_mismatch PROPERTIES WORKING_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR})
SET_TESTS_PROPERTIES(md5_will_sum_mismatch PROPERTIES PASS_REGULAR_EXPRESSION "test1.txt: FAILED")

ADD_TEST(md5_simple ${PROJECT_BINARY_DIR}/md5sum -c test1.md5)
SET_TESTS_PROPERTIES(md5_simple PROPERTIES WORKING_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR})

ADD_TEST(md5_with_space ${PROJECT_BINARY_DIR}/md5sum -c "test with spaces.md5")
SET_TESTS_PROPERTIES(md5_with_space PROPERTIES WORKING_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR})

ADD_TEST(md5_file_missing ${PROJECT_BINARY_DIR}/md5sum -c test_file_missing.md5)
SET_TESTS_PROPERTIES(md5_file_missing PROPERTIES WORKING_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR})
SET_TESTS_PROPERTIES(md5_file_missing PROPERTIES PASS_REGULAR_EXPRESSION "test2.txt: File not found")

ADD_TEST(md5_with_asterisk ${PROJECT_BINARY_DIR}/md5sum -c test_with_asterisk.md5)
SET_TESTS_PROPERTIES(md5_with_asterisk PROPERTIES WORKING_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR})

ADD_TEST(md5_create_sum ${PROJECT_BINARY_DIR}/md5sum test1.txt)
SET_TESTS_PROPERTIES(md5_create_sum PROPERTIES WORKING_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR})
SET_TESTS_PROPERTIES(md5_create_sum PROPERTIES PASS_REGULAR_EXPRESSION "a5890ace30a3e84d9118196c161aeec2  test1.txt")
